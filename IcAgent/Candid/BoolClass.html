def decode_value(b, t)
  check_type(t)
  byte = IcAgent::Candid.safe_read_byte(b)
  str_io = StringIO.new
  str_io.putc(byte.hex)
  if LEB128.decode_signed(str_io) == 1
    true
  elsif LEB128.decode_signed(str_io) == 0
    false
  else
    raise ValueError, 'Boolean value out of range'
  end
end
