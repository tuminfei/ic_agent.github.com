def _build_type_table_impl(type_table)
  if @type.nil?
    raise 'Recursive type uninitialized'
  else
    if !get_type.is_a?(PrimitiveType)
      type_table.add(self, '')
      @type.build_type_table(type_table)
      type_table.merge(self, @type.name)
    end
  end
end
